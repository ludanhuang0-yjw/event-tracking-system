getFilteredEvents() {

const searchTerm = document.getElementById('searchInput').value.toLowerCase();

const platformFilter = document.getElementById('platformFilter').value;

const sortBy = document.getElementById('sortBy').value;


let filtered = this.events.filter(event => {

const latest = event.versions[event.versions.length - 1];

const matchSearch = !searchTerm ||

event.name.toLowerCase().includes(searchTerm) ||

event.nameZh?.toLowerCase().includes(searchTerm) ||

latest.desc?.toLowerCase().includes(searchTerm) ||

latest.parameters?.some(p =>

p.name.toLowerCase().includes(searchTerm) ||

p.desc.toLowerCase().includes(searchTerm)

);

const matchPlatform = !platformFilter || event.platform === platformFilter;

return matchSearch && matchPlatform;

});


filtered.sort((a, b) => {

const latestA = a.versions[a.versions.length - 1];

const latestB = b.versions[b.versions.length - 1];

if (sortBy === 'newest') return new Date(latestA.createdAt) - new Date(latestB.createdAt);

if (sortBy === 'updated') return new Date(latestB.createdAt) - new Date(latestA.createdAt);

if (sortBy === 'name') return a.name.localeCompare(b.name);

return 0;

});


return filtered;

}

formatDate(dateStr) {

const date = new Date(dateStr);

const now = new Date();

const diff = now - date;

if (diff < 60000) return '刚刚';

if (diff < 3600000) return `${Math.floor(diff/60000)}分钟前`;

if (diff < 86400000) return `${Math.floor(diff/3600000)}小时前`;

if (diff < 604800000) return `${Math.floor(diff/86400000)}天前`;

return date.toLocaleDateString('zh-CN');

}

// 版本对比

compareWithPrevious() {

const event = this.events.find(e => e.id === this.currentEventId);

const versions = [...event.versions].reverse();

const current = versions[this.currentVersionIndex];

const previous = versions[this.currentVersionIndex + 1];

if (!previous) {

this.showToast('没有更早的版本', 'error');

return;

}

const diffContainer = document.getElementById('diffContent');

const diffView = document.getElementById('diffView');

diffView.classList.remove('hidden');

let diffHtml = '';

// 对比描述

if (current.desc !== previous.desc) {

diffHtml += `

<div class="p-3 rounded-lg border">

<p class="text-xs font-medium text-slate-500 mb-1">描述变更</p>

<div class="grid grid-cols-2 gap-2 text-sm">

<div class="diff-removed p-2 rounded line-through opacity-70">${previous.desc || '无'}</div>

<div class="diff-added p-2 rounded">${current.desc || '无'}</div>

</div>

</div>

`;

}

// 对比平台

if (current.platform !== previous.platform) {

diffHtml += `

<div class="p-3 rounded-lg border">

<p class="text-xs font-medium text-slate-500 mb-1">平台变更</p>

<div class="flex gap-4 text-sm">

<span class="diff-removed px-2 py-1 rounded">${previous.platform}</span>

<span class="text-slate-400">→</span>

<span class="diff-added px-2 py-1 rounded">${current.platform}</span>

</div>

</div>

`;

}

// 对比参数

const prevParams = previous.parameters || [];

const currParams = current.parameters || [];

const addedParams = currParams.filter(cp => !prevParams.find(pp => pp.name === cp.name));

const removedParams = prevParams.filter(pp => !currParams.find(cp => cp.name === pp.name));

const changedParams = currParams.filter(cp => {

const pp = prevParams.find(p => p.name === cp.name);

return pp && (pp.type !== cp.type || pp.desc !== cp.desc);

});

if (addedParams.length > 0 || removedParams.length > 0 || changedParams.length > 0) {

diffHtml += `<div class="p-3 rounded-lg border"><p class="text-xs font-medium text-slate-500 mb-2">参数变更</p><div class="space-y-2">`;

addedParams.forEach(p => {

diffHtml += `

<div class="diff-added p-2 rounded text-sm flex items-center gap-2">

<span class="font-mono font-medium">+ ${p.name}</span>

<span class="text-xs bg-white px-1 rounded">${p.type}</span>

<span class="text-slate-600">${p.desc}</span>

</div>

`;

});

removedParams.forEach(p => {

diffHtml += `

<div class="diff-removed p-2 rounded text-sm flex items-center gap-2 line-through opacity-70">

<span class="font-mono font-medium">- ${p.name}</span>

<span class="text-xs bg-white px-1 rounded">${p.type}</span>

</div>

`;

});

changedParams.forEach(p => {

const oldP = prevParams.find(op => op.name === p.name);

diffHtml += `

<div class="diff-changed p-2 rounded text-sm">

<div class="flex items-center gap-2 mb-1">

<span class="font-mono font-medium">~ ${p.name}</span>

<span class="text-xs text-slate-500">有修改</span>

</div>

<div class="grid grid-cols-2 gap-2 text-xs pl-4">

<div class="line-through opacity-60">${oldP.type} | ${oldP.desc}</div>

<div>${p.type} | ${p.desc}</div>

</div>

</div>

`;

});

diffHtml += `</div></div>`;

}

if (diffHtml === '') {

diffHtml = '<p class="text-slate-500 text-center py-4">与上一版本相比无变更</p>';

}

diffContainer.innerHTML = diffHtml;

}


// 回滚版本

restoreVersion() {

if (!confirm('确定要回滚到此版本吗？这将创建一个新的版本记录。')) return;

const event = this.events.find(e => e.id === this.currentEventId);

const versions = [...event.versions].reverse();

const targetVersion = versions[this.currentVersionIndex];

// 创建新版本，内容复制自历史版本

const newVersion = {

...targetVersion,

version: this.incrementVersion(event.currentVersion),

changeLog: `从 v${targetVersion.version} 回滚`,

createdAt: new Date().toISOString(),

createdBy: 'current_user'

};

event.versions.push(newVersion);

event.currentVersion = newVersion.version;

this.saveData();

this.renderVersionHistory(event);

this.selectVersion(0); // 选中最新版本

this.showToast('已回滚并创建新版本');

}


incrementVersion(version) {

const parts = version.split('.').map(Number);

parts[2] = (parts[2] || 0) + 1;

return parts.join('.');

}

saveEvent(e) {

e.preventDefault();

const eventId = document.getElementById('editEventId').value;

const event = this.events.find(ev => ev.id === eventId);

const newVersion = document.getElementById('editEventVersion').value.trim();

// 验证版本号

if (!newVersion) {

this.showToast('请输入新版本号', 'error');

return;

}

// 检查版本号是否递增

const versions = event.versions.map(v => v.version);

if (versions.includes(newVersion)) {

this.showToast('版本号已存在', 'error');

return;

}

// 收集参数

const parameters = [];

document.querySelectorAll('#editParametersList > div').forEach(row => {

parameters.push({

name: row.querySelector('.param-name').value.trim(),

type: row.querySelector('.param-type').value,

desc: row.querySelector('.param-desc').value.trim(),

example: row.querySelector('.param-example').value.trim()

});

});

const versionData = {

version: newVersion,

desc: document.getElementById('editEventDesc').value.trim(),

platform: document.getElementById('editEventPlatform').value,

parameters: parameters,

changeLog: document.getElementById('editChangeLog').value.trim(),

createdAt: new Date().toISOString(),

createdBy: 'current_user'

};

event.versions.push(versionData);

event.currentVersion = newVersion;

event.nameZh = document.getElementById('editEventNameZh').value.trim();

this.saveData();

this.closeEditModal();

// 如果在详情页，刷新详情

if (this.currentEventId === eventId) {

this.renderVersionHistory(event);

this.selectVersion(0);

} else {

this.renderList();

}

this.showToast('新版本已保存');

createNewEvent(e) {

e.preventDefault();

const name = document.getElementById('newEventName').value.trim();

// 检查名称唯一性

if (this.events.find(ev => ev.name === name)) {

this.showToast('Event 名称已存在', 'error');

return;

}

const parameters = [];

document.querySelectorAll('#createParametersList > div').forEach(row => {

parameters.push({

name: row.querySelector('.param-name').value.trim(),

type: row.querySelector('.param-type').value,

desc: row.querySelector('.param-desc').value.trim(),

example: row.querySelector('.param-example').value.trim()

});

});

const newEvent = {

id: 'evt_' + Date.now(),

name: name,

nameZh: document.getElementById('newEventNameZh').value.trim(),

currentVersion: document.getElementById('newEventVersion').value.trim(),

platform: document.getElementById('newEventPlatform').value,

versions: [{

version: document.getElementById('newEventVersion').value.trim(),

desc: document.getElementById('newEventDesc').value.trim(),

platform: document.getElementById('newEventPlatform').value,

parameters: parameters,

changeLog: '初始版本创建',

createdAt: new Date().toISOString(),

createdBy: 'current_user'

}]

};

this.events.unshift(newEvent);

this.saveData();

this.closeCreateModal();

this.renderList();

this.showToast('埋点创建成功');

}


// 删除

deleteEvent(eventId) {

const event = this.events.find(e => e.id === eventId);

if (event) {

this.deleteEventId = eventId;

document.getElementById('deleteEventName').textContent = event.name;

document.getElementById('deleteModal').classList.remove('hidden');

}

}


closeDeleteModal() {

document.getElementById('deleteModal').classList.add('hidden');

this.deleteEventId = null;

}


confirmDelete() {

if (this.deleteEventId) {

this.events = this.events.filter(e => e.id !== this.deleteEventId);

this.saveData();

this.renderList();

this.closeDeleteModal();

this.showToast('埋点已删除');

}

}


// 导入导出

exportData() {

const dataStr = JSON.stringify(this.events, null, 2);

const blob = new Blob([dataStr], { type: 'application/json' });

const url = URL.createObjectURL(blob);

const a = document.createElement('a');

a.href = url;

a.download = `tracking-events-${new Date().toISOString().split('T')[0]}.json`;

a.click();

URL.revokeObjectURL(url);

this.showToast('数据已导出');

}


importData(input) {

const file = input.files[0];

if (!file) return;

const reader = new FileReader();

reader.onload = (e) => {

try {

const data = JSON.parse(e.target.result);

if (Array.isArray(data)) {

// 合并数据

let added = 0;

data.forEach(item => {

if (!this.events.find(ev => ev.name === item.name)) {

this.events.push(item);

added++;

}

});

this.saveData();

this.renderList();

this.showToast(`成功导入 ${added} 个埋点`);

}

} catch (err) {

this.showToast('文件格式错误', 'error');

}

input.value = '';

};

reader.readAsText(file);

}


showToast(message, type = 'success') {

const toast = document.getElementById('toast');

document.getElementById('toastMessage').textContent = message;

document.getElementById('toastIcon').textContent = type === 'success' ? '✓' : '✕';

toast.className = `fixed bottom-6 right-6 z-50 transform transition-all duration-300 ${type === 'success' ? 'bg-slate-800' : 'bg-red-600'} text-white px-6 py-3 rounded-xl shadow-lg flex items-center gap-3`;

setTimeout(() => toast.classList.remove('translate-y-20', 'opacity-0'), 10);

setTimeout(() => toast.classList.add('translate-y-20', 'opacity-0'), 3000);

}

}
